// A reusable component for a pack card with its own quantity state and functions.
const PackCard = ({ title, description, price }) => {
  const [quantity, setQuantity] = useState(1);

  // Function to increment the pack quantity
  const increment = () => {
    setQuantity((prev) => prev + 1);
  };

  // Function to decrement the pack quantity (ensuring at least 1 is selected)
  const decrement = () => {
    setQuantity((prev) => (prev > 1 ? prev - 1 : 1));
  };
  

  return (
    <Box
      w={"480px"}
      h={"350px"}
      bg={"white"}
      shadow={"md"}
      borderRadius={"16px"}
      display={"flex"}
      flexDir={"column"}
      alignItems={"start"}
      justifyContent={"start"}
    >
      <Box
        w={"100%"}
        h={"500px"}
        bg={"gray.100"}
        borderRadius={"16px"}
        borderBottomRadius={"0px"}
      ></Box>

      <Box p={4} w={"100%"} h={"100%"}>
        <HStack align={"center"} gap={2}>
          <Package />
          <Text fontWeight={"semibold"} fontSize={"18px"}>
            {title}
          </Text>
        </HStack>
        <Box mt={2}>
          <Text fontSize={"16px"} color={"gray.600"}>
            {description}
          </Text>
        </Box>
      </Box>

      <HStack
        p={4}
        w={"100%"}
        gap={2}
        mt={8}
        align={"center"}
        justify={"space-between"}
      >
        <Text fontWeight={"semibold"} fontSize={"32px"}>
          ${price}
        </Text>
        <HStack>
          <Box
            as={"button"}
            p={1}
            border={"1px solid lightgray"}
            rounded={"6px"}
            onClick={increment}
          >
            <Plus />
          </Box>
          <Box
            as={"button"}
            p={1}
            rounded={"6px"}
            fontWeight={"semibold"}
            fontSize={"18px"}
          >
            {quantity}
          </Box>
          <Box
            as={"button"}
            p={1}
            border={"1px solid lightgray"}
            rounded={"6px"}
            onClick={decrement}
          >
            <Minus />
          </Box>
        </HStack>
      </HStack>
    </Box>
  );
};
